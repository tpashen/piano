# piano

main.py – головний файл, у якому запускається гра.
settings.py – налаштування гри (розмір вікна, кольори, список нот).
sounds.py – робота зі звуками (завантаження з assets/sounds/).
keys.py – клавіші піаніно та їхня анімація.
buttons.py – кнопки меню («Settings»).
effects.py – ефекти для клавіш (наприклад, підсвітка при натисканні).
